IMPROVE RECOMMENDATION MATCHING LOGIC:

Replace the current static recommendation display with 
this scoring function that picks the best 5 destinations 
from all 20 based on quiz answers:

function getRecommendations(quizAnswers) {
  const { mood, companion, budget, activity, interests, city } = quizAnswers
  
  const budgetMap = {
    'under-15k': 15000,
    '15-30k': 30000,
    '30-60k': 60000,
    '60k+': 999999
  }
  const maxBudget = budgetMap[budget] || 30000
  
  const scored = destinations.map(dest => {
    let score = 0
    
    // Budget match (0-30 points)
    if (dest.totalBudget <= maxBudget) score += 30
    else if (dest.totalBudget <= maxBudget * 1.2) score += 15
    
    // Companion match (0-25 points)
    if (dest.companionTypes.includes(companion)) score += 25
    
    // Activity level match (0-20 points)
    if (dest.activityLevel === activity) score += 20
    else if (activity === 'moderate' && dest.activityLevel === 'low') score += 10
    
    // Interest match (0-25 points)
    if (interests && interests.length > 0) {
      const tagMatch = dest.tags.some(tag => 
        interests.some(interest => 
          tag.toLowerCase().includes(interest.toLowerCase()) ||
          interest.toLowerCase().includes(tag.toLowerCase())
        )
      )
      if (tagMatch) score += 25
    }
    
    // Season bonus (0-10 points)
    if (dest.currentSeason === 'ideal') score += 10
    
    return { ...dest, score }
  })
  
  return scored
    .sort((a, b) => b.score - a.score)
    .slice(0, 5)
}
