const ANTHROPIC_API_KEY = import.meta.env.VITE_ANTHROPIC_API_KEY

export async function callClaude(
  systemPrompt: string,
  messages: { role: 'user' | 'assistant', content: string }[],
  maxTokens: number = 1000
): Promise<string> {
  
  const response = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'x-api-key': ANTHROPIC_API_KEY,
      'anthropic-version': '2023-06-01',
      'anthropic-dangerous-direct-browser-access': 'true'
    },
    body: JSON.stringify({
      model: 'claude-opus-4-5',
      max_tokens: maxTokens,
      system: systemPrompt,
      messages: messages
    })
  })

  if (!response.ok) {
    const error = await response.json()
    throw new Error(`Claude API error: ${error.error?.message}`)
  }

  const data = await response.json()
  return data.content[0].text
}

function parseClaudeResponse(response: string) {
  const destinationsMatch = response.match(
    /DESTINATIONS:\s*(\[[\s\S]*?\])/
  )
  
  let destinations = []
  let text = response
  
  if (destinationsMatch) {
    try {
      destinations = JSON.parse(destinationsMatch[1])
      text = response
        .replace(/DESTINATIONS:\s*\[[\s\S]*?\]/, '')
        .trim()

  }
