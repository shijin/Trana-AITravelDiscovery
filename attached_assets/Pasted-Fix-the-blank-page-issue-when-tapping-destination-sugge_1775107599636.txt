Fix the blank page issue when tapping destination 
suggestions returned by the Claude chat response.

PROBLEM: When Claude recommends a destination in chat 
that does not exist in the mockData destinations array, 
tapping it navigates to a blank destination detail page 
because no data is found.

FIX 1 — Update the destination tap handler in the 
chat screen.

When a user taps a destination card returned by Claude,
use this logic instead of directly navigating:

function handleDestinationTap(destination: {
  name: string, 
  state: string, 
  reason: string
}) {
  // First try to find in mockData by name match
  const existingDestination = destinations.find(d => 
    d.name.toLowerCase() === destination.name.toLowerCase() ||
    d.name.toLowerCase().includes(
      destination.name.toLowerCase()
    ) ||
    destination.name.toLowerCase().includes(
      d.name.toLowerCase()
    )
  )
  
  if (existingDestination) {
    // Navigate to existing destination detail
    navigate(`/destination/${existingDestination.id}`, {
      state: { destination: existingDestination }
    })
  } else {
    // Build a dynamic destination object from 
    // Claude's response data
    const dynamicDestination = {
      id: destination.name.toLowerCase()
        .replace(/\s+/g, '-'),
      name: destination.name,
      state: destination.state,
      region: destination.state,
      heroGradient: getGradientForDestination(
        destination.name
      ),
      tags: getTagsFromReason(destination.reason),
      rationale: destination.reason,
      budgetBreakdown: { 
        transport: 5000, 
        stay: 8000, 
        food: 3000 
      },
      totalBudget: 16000,
      bestMonths: ["Oct","Nov","Dec","Jan","Feb","Mar"],
      currentSeason: "ideal",
      activityLevel: "moderate",
      companionTypes: ["solo","couple","family"],
      foodHighlights: [
        `Local ${destination.state} cuisine`,
        "Regional specialties",
        "Street food"
      ],
      videos: [],
      isDynamic: true
    }
    
    navigate(
      `/destination/${dynamicDestination.id}`, 
      { state: { destination: dynamicDestination } }
    )
  }
}

// Helper function to assign gradient based on 
// destination name
function getGradientForDestination(name: string): string {
  const gradients = [
    "from-green-700 to-emerald-500",
    "from-blue-700 to-cyan-500",
    "from-purple-700 to-indigo-500",
    "from-teal-700 to-green-500",
    "from-orange-700 to-amber-500",
    "from-rose-700 to-pink-500",
    "from-slate-700 to-gray-500",
    "from-violet-700 to-purple-500"
  ]
  // Deterministic selection based on name length
  const index = name.length % gradients.length
  return gradients[index]
}

// Helper to extract tags from reason text
function getTagsFromReason(reason: string): string[] {
  const tagMap: Record<string, string[]> = {
    'beach': ['Beach', 'Coastal', 'Scenic'],
    'mountain': ['Mountains', 'Trekking', 'Adventure'],
    'heritage': ['Heritage', 'History', 'Culture'],
    'nature': ['Nature', 'Wildlife', 'Outdoors'],
    'spiritual': ['Spiritual', 'Temple', 'Peaceful'],
    'food': ['Food', 'Culinary', 'Local Culture'],
    'waterfall': ['Nature', 'Scenic', 'Offbeat'],
    'forest': ['Nature', 'Offbeat', 'Adventure'],
    'lake': ['Scenic', 'Nature', 'Peaceful'],
    'temple': ['Spiritual', 'Heritage', 'Culture']
  }
  
  const reasonLower = reason.toLowerCase()
  for (const [keyword, tags] of Object.entries(tagMap)) {
    if (reasonLower.includes(keyword)) return tags
  }
  return ['Offbeat', 'Explore', 'Discover']
}
