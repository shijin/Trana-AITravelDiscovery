trip requests and build combined itineraries.

FIX 2 — UPDATE RESPONSE PARSER
═══════════════════════════════════════════

Update the parseClaudeResponse function to 
handle both intent types:

function parseClaudeResponse(response: string) {
  // Extract the RESPONSE JSON block
  const responseMatch = response.match(
    /RESPONSE:\s*(\{[\s\S]*?\})\s*$/
  )
  
  if (!responseMatch) {
    // Fallback for old format
    return {
      intent: 'single',
      text: response,
      destinations: []
    }
  }
  
  try {
    const parsed = JSON.parse(responseMatch[1])
    return parsed
  } catch {
    return {
      intent: 'single', 
      text: response
        .replace(/RESPONSE:\s*\{[\s\S]*\}/, '')
        .trim(),
      destinations: []
    }
  }
}

FIX 4 — BUILD COMBINED CIRCUIT HANDLER
═══════════════════════════════════════════

Add this function to the Chat screen:

async function handleBuildCombinedCircuit(
  destinations: Array<{
    name: string, 
    state: string, 
    suggestedDays: number,
    reason: string
  }>,
  totalDays: number,
  tripTitle: string
) {
  setIsBuilding(true)
  
  try {
    const quizAnswers = location.state?.quizAnswers || {}
    const startCity = quizAnswers.city || 'your city'
    const budget = quizAnswers.budget || '₹30,000–₹60,000'
    const companion = quizAnswers.companion || 'traveler'
    
    const destinationList = destinations
      .map(d => `${d.name} (${d.suggestedDays} days)`)
      .join(', ')
    
    const userPrompt = 
`Build a ${totalDays}-day travel circuit covering 
these destinations in order:
${destinationList}

Traveler context:
- Traveling with: ${companion}
- Budget: ${budget} total for the trip
- Starting from: ${startCity}

IMPORTANT: 
- Include ALL destinations listed
- Give exactly the number of days specified 
  for each destination
- Total days must equal ${totalDays}
- Plan travel between destinations logically`

    const response = await callClaude(
      ITINERARY_SYSTEM_PROMPT,
      [{ role: 'user', content: userPrompt }],
      2500
    )
    
    const itinerary = parseItinerary(response)
    
    if (itinerary) {
      navigate('/itinerary/detail', {
        state: { 
          itinerary,
          tripTitle 
        }
      })
    } else {
      showToast(
        'Could not build the circuit. Please try again.',
        'error'
      )
    }
  } catch (error) {
    showToast(
      'Something went wrong. Please try again.',
      'error'
    )
  } finally {
    setIsBuilding(false)
  }
}
